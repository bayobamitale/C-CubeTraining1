{

	declare a variable to store minimum number: minimumNumber;

CalaculteAverage() 
{
	read an array of numbers from user
	declare a variable to store the count of elements in numbers countOfElementsInArray
	initialize the countOfElementsInArray to zero
	declare a variable to store the sum of elements of numbers: sumOfElementsInArray;
	initialize the variable to zero
	
	loop through the array using foreach construct:
		foreach( variable number inside array numbers)
		{
				add number to sumOfElementsInArray
				store the result back into sumOfElementsInArray
				increment countOfElementsInArray by 1
		}
		
		print or return sumOfElementsInArray/countOfElementsInArray
}
